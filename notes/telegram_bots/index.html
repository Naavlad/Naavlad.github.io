Структура callbasks 
# bot/callbasks/<name.py>

from aiogram import types
from aiogram.dispatcher import FSMContext
from aiogram.dispatcher.filters.state import State, StatesGroup

from bot.init_bot import dp


class WalletState(StatesGroup):
    input_name = State()
    input_description = State()


@dp.callback_query_handler(text="wallets_create")
async def start_state_input_name(call: types.CallbackQuery):
    """Handler callback start State

    Начало создания Wallet
        Вход в WalletState
        Получение name : input_name
    """
    await call.message.edit_reply_markup()
    await call.message.answer("Введите:")
    await WalletState.input_name.set()


@dp.message_handler(state=WalletState.input_name)
async def state_input_description(
    message: types.Message, state: FSMContext
):
    """State WalletState next input_name

    Получаем name из message.text
    Запись name в state
    Получение description : input_description
    """
    # записываем полученное название группы в state
    wallet_name = message.text
    async with state.proxy() as data:
        data["wallet_name"] = wallet_name

    await message.answer("Введите описание:")
    await WalletState.input_description.set()


@dp.message_handler(state=WalletState.input_description)
async def finish_state(message: types.Message, state: FSMContext):
    """State WalletState next input_description

    Достаем name из state
    Получаем description из message.text
    создаем кошелек
    закрываем state : state.finish
    """
    # Получаем название кошелька из state
    data = await state.get_data()
    wallet_name = data.get("wallet_name")

    wallet_description = message.text
    wallet_owner_id = message.from_user.id

    wallet = WalletCRUD.create(
        name=wallet_name,
        description=wallet_description,
        owner_id=wallet_owner_id,
    )
    if wallet:
        await message.answer(f"✅ Кошелек создан.")
        await state.finish()
        logger.info(f"Оk. Wallet created. {wallet.id}")

    else:
        await message.answer(f"❗ Что то пошло не так...")
        await state.finish()
        logger.info("Error Wallet created.")
